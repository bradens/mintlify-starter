<ResponseField name="buySuccess" type="Boolean">

<ResponseField name="buyGasUsed" type="String">

</ResponseField>

<ResponseField name="maxBuyAmount" type="String">

</ResponseField>

<ResponseField name="sellSuccess" type="Boolean">
  Whether or not a token was able to be succesfully sold during simulation.
</ResponseField>

<ResponseField name="sellTax" type="String">
  Tax paid for a sell transaction during simulation.
</ResponseField>

<ResponseField name="sellGasUsed" type="String">
  Gas used for a sell transaction during simulation.
</ResponseField>

<ResponseField name="maxSellAmount" type="String">
  The maximum token amount an address can sell during simulation.
</ResponseField>

<ResponseField name="canTransferOwnership" type="Boolean">
  Whether or not the contract ownership was able to be transferred during simulation.
</ResponseField>

<ResponseField name="canRenounceOwnership" type="Boolean">
  Whether or not the contract ownership was able to be renounced during simulation.
</ResponseField>

<ResponseField name="isOwnerRenounced" type="Boolean">
  Whether or not the contract ownership is already renounced during simulation (owner is 0x0).
</ResponseField>

<ResponseField name="openTradingCall" type="String">
  If a call was found to trigger liquidity & trading, this is the call name.
</ResponseField>
