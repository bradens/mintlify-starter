### Arguments

<ResponseField name="limit" type="Int">
  The maximum number of events to return.
</ResponseField>

<a href="/api-reference/input-objects#makereventsqueryinput">
  <ResponseField name="query" type="MakerEventsQueryInput!" required>
    The query filters to apply to the results.
  </ResponseField>
</a>

<ResponseField name="cursor" type="String">
  The cursor to use for pagination.
</ResponseField>

<a href="/api-reference/enums#rankingdirection">
  <ResponseField name="direction" type="RankingDirection">
    The order to receive the token events. Can be `DESC` (default) or `ASC`.
  </ResponseField>
</a>
