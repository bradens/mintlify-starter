<ResponseField name="poolAddress" type="String">

</ResponseField>

<ResponseField name="networkId" type="Int">
  The network ID to filter by.
</ResponseField>

<ResponseField name="collectionAddress" type="String">
  The contract address of the NFT collection to filter by.
</ResponseField>

<ResponseField name="exchangeAddress" type="String">
  The contract address of the NFT AMM marketplace to filter by.
</ResponseField>
